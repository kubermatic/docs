{{- /* Returns the count of non-excluded pages with content under a section.
       Context: a section page. */ -}}
{{- $count := 0 -}}
{{- range .Pages -}}
  {{- if not (partial "llms-is-excluded.txt" .) -}}
    {{- if .Plain -}}
      {{- $count = add $count 1 -}}
    {{- end -}}
    {{- if .IsSection -}}
      {{- $count = add $count (partial "llms-page-count.txt" .) -}}
    {{- end -}}
  {{- end -}}
{{- end -}}
{{- return $count -}}
